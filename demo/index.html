    Basic example

    $("video").bootstrapVideo(); Scroll top

    Disable play/pause on click

    $("video").bootstrapVideo({ playpauseonclick: false }); Scroll top

    Set initial volume

    $("video").bootstrapVideo({ volume: 0.5 }); Scroll top

    Mute on init

    $("video").bootstrapVideo({ muted: true }); Scroll top

    Enable autoplay

    $("video").bootstrapVideo({ autoplay: true }); Scroll top

    Hide all controls

    $("video").bootstrapVideo({ controls: false }); Scroll top

    Disable scrolling

    $("video").bootstrapVideo({ draggableseekbar: false }); Scroll top

    Set custom size

    $("video").bootstrapVideo({ width: "300px" }); Scroll top

    Repeat when finished

    $("video").bootstrapVideo({ repeatonstop: true }); Scroll top

    Set custom class for paused video

    $("video").bootstrapVideo({ pauseclass: "happy" }); Scroll top

    Hide play/pause button

    $("video").bootstrapVideo({ playpausebutton: false }); Scroll top

    Hide mute button

    $("video").bootstrapVideo({ mutebutton: false }); Scroll top

    Hide progress bar

    $("video").bootstrapVideo({ seekbar: false }); Scroll top

    Hide volume button

    $("video").bootstrapVideo({ volumebutton: false }); Scroll top
